% NATURAL LOGIC
%%%%%%%%%%%%%%%%%%%
\def\title{Natural Logic}
\begin{frame}{\title}
\begin{center}
\hh{If I mutate a sentence in this way, do I preserve its truth?}
\end{center}
\vspace{1em}
\pause

\hh{Braindead for humans, but not computers}
\begin{itemize}
  \item \w{All young rabbits drink milk}
        $\centernot \implies$ \w{All rabbits drink milk} \\
        \w{Some young rabbits drink milk}
        $\implies$ \w{Some rabbits drink milk} \\
%  \pause
%  \item \w{Enemies give fake praise}
%        $\centernot \implies$ \w{Enemies give praise} \\
%        \w{Friends give frequent praise}
%        $\implies$ \w{Friends give praise} \\
\end{itemize}
\vspace{2em}
\pause

\hh{Hard even for first order logic}
\begin{itemize}
  \item \w{Most cats eat mice}
        $\implies$ \w{Most cats eat rodents} \\
  \item \w{All students who know a foreign language learned it at university} \\
        $\implies$ \w{They learned it at school.}
\end{itemize}
\end{frame}
