\begin{center}
  \w{All young rabbits drink milk} \\ 
  $\centernot \implies$ \w{All rabbits drink milk}
\end{center}

\hh{Step 2:} Shorten each clause:
\begin{itemize}
  \item[]
      \begin{dependency}[text only label, label style={above}]
        \begin{deptext}[column sep=-0.05cm]
          she \& Born \& in \& a \& small \& town \\
        \end{deptext}
        \depedge[edge unit distance=1.25ex]{2}{6}{prep\_in}
        \depedge[edge unit distance=1.0ex]{6}{5}{amod}
        \depedge[edge unit distance=1.4ex]{6}{4}{det}
        \depedge[edge unit distance=2.0ex, edge style={blue!60!black,thick}]{2}{1}{\darkblue{nsubj}}
      \end{dependency}
  \item[] $\implies$ \w{She born in a small town}
  \item[] $\implies$ \w{She born in small town}
  \item[] $\implies$ \w{She born in town}
\end{itemize}
\end{frame}

% Natural Logic
%%%%%%%%%%%%%%%%%%%
\begin{frame}{Natural Logic}
\begin{center}
\hh{If I mutate a sentence in this way, do I preserve its truth?}
\end{center}
\vspace{1em}
\pause

\hh{Braindead for humans, but not computers}
\begin{itemize}
  \item \w{All young rabbits drink milk}
        $\centernot \implies$ \w{All rabbits drink milk} \\
        \w{Some young rabbits drink milk}
        $\implies$ \w{Some rabbits drink milk} \\
  \pause
  \item \w{Enemies give fake praise}
        $\centernot \implies$ \w{Enemies give praise} \\
        \w{Friends give frequent praise}
        $\implies$ \w{Friends give praise} \\
\end{itemize}
\vspace{1em}
\pause

\hh{Hard even for first order logic}
\begin{itemize}
  \item \w{Most cats eat mice}
        $\implies$ \w{Most cats eat rodents} \\
  \item \w{All students who know a foreign language learned it at university} \\
        $\implies$ \w{They learned it at school.}
\end{itemize}
\end{frame}
