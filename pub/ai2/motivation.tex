\begin{frame}{Machine Reading Requires Logic}

\hh{Textual entailment}
\begin{itemize}
\item \w{All cats have tails} $\implies$ \w{Some cats have tails}.
\item \w{A soccer game with multiple males playing} $\implies$ \w{Some men are playing a sport}.
\end{itemize}
\vspace{1ex}
\pause

\hh{Sentiment analysis}
\begin{itemize}
\item \w{That movie was \textbf{not} the best.}
\item \w{Some parts were slow} / \w{The whole movie was slow.}
\end{itemize}
\vspace{1ex}
\pause

\hh{Relation extraction}
\begin{itemize}
\item \w{Some people think that Obama was born in Kenya.}
\item \w{Barack Obama was \textbf{not} born in Kenya.}
\end{itemize}

\end{frame}

% UNEXPRESSIVE
%%%%%%%%%%%%%%%%%%%
\begin{frame}{First Order Logic is \textbf{Unexpressive}}
\begin{center}
  \includegraphics[height=2cm]{../img/pokerface.png}
\end{center}

\hh{\textit{Some people think that Obama was born in Kenya.}}
\begin{itemize}
\item Second order logic: 
      $\exists x \exists P \left[ P = \textrm{born} \land \textbf{\textrm{think}(x, P)} \land P(\textrm{Obama}, \textrm{Kenya}) \right]$
\pause
\item But, can still infer: \w{Some people think that Obama is from Kenya.}
\end{itemize}
\vspace{1ex}
\pause


\hh{\textit{Most students who learned a foreign language learned it at a university.}}
\begin{itemize}
\item \w{Most} is not a first-order quantifier.
\item Scoping ambiguities everywhere!
\pause
\item But, can still infer: \w{Most students learned it at a school.}
\end{itemize}

\end{frame}
